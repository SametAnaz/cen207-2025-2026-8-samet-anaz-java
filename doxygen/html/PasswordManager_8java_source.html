741 }

918 private void heapSort(ServiceUsage[] arr) {
919 int n = arr.length;
920
921 // Build max heap
922 for (int i = n / 2 - 1; i >= 0; i--) {
923 heapify(arr, n, i);
924 }
925
926 // Extract elements from heap one by one
927 for (int i = n - 1; i > 0; i--) {
928 // Move current root to end
929 ServiceUsage temp = arr[0];
930 arr[0] = arr[i];
931 arr[i] = temp;
932 // Heapify the reduced heap
933 heapify(arr, i, 0);
934 }
935 }

944 private void heapify(ServiceUsage[] arr, int n, int i) {
945 int largest = i;
946 int left = 2 * i + 1;
947 int right = 2 * i + 2;
948
949 // Check if left child is larger than root
950 if (left < n && arr[left].compareTo(arr[largest]) > 0) {
951 largest = left;
952 }
953
954 // Check if right child is larger than current largest
955 if (right < n && arr[right].compareTo(arr[largest]) > 0) {
956 largest = right;
957 }
958
959 // If largest is not root
960 if (largest != i) {
961 ServiceUsage swap = arr[i];
962 arr[i] = arr[largest];
963 arr[largest] = swap;
964 // Recursively heapify the affected subtree
965 heapify(arr, n, largest);
966 }
