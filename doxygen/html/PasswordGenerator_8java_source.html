81 public static String generatePassword(int length, boolean includeUppercase,
82 boolean includeLowercase, boolean includeDigits,
83 boolean includeSpecial) {
84 // Return empty string for zero or negative length
85 if (length <= 0) {
86 return "";
87 }
88
89 // Build the character set based on includes
90 StringBuilder charSetBuilder = new StringBuilder();
91
92 if (includeUppercase) charSetBuilder.append(UPPERCASE_CHARS);
93
94 if (includeLowercase) charSetBuilder.append(LOWERCASE_CHARS);
95
96 if (includeDigits) charSetBuilder.append(DIGIT_CHARS);
97
98 if (includeSpecial) charSetBuilder.append(SPECIAL_CHARS);
99
100 // If no character set is selected, use lowercase as default
101 String charSet = charSetBuilder.length() > 0 ? charSetBuilder.toString() : LOWERCASE_CHARS;
102 // Generate the password
103 StringBuilder password = new StringBuilder(length);
104 Random random = new Random();
105
106 // Ensure at least one character from each selected char set
107 if (length >= 1 && includeUppercase && password.length() < length) {
108 password.append(UPPERCASE_CHARS.charAt(random.nextInt(UPPERCASE_CHARS.length())));
109 }
110
111 if (length >= 2 && includeLowercase && password.length() < length) {
112 password.append(LOWERCASE_CHARS.charAt(random.nextInt(LOWERCASE_CHARS.length())));
113 }
114
115 if (length >= 3 && includeDigits && password.length() < length) {
116 password.append(DIGIT_CHARS.charAt(random.nextInt(DIGIT_CHARS.length())));
117 }
118
119 if (length >= 4 && includeSpecial && password.length() < length) {
120 password.append(SPECIAL_CHARS.charAt(random.nextInt(SPECIAL_CHARS.length())));
121 }
122
123 // Fill the rest with random characters from the selected set
124 while (password.length() < length) {
125 int index = random.nextInt(charSet.length());
126 password.append(charSet.charAt(index));
127 }
128
129 // Shuffle the characters to avoid predictable patterns
130 char[] passChars = password.toString().toCharArray();
131
132 for (int i = passChars.length - 1; i > 0; i--) {
133 int j = random.nextInt(i + 1);
134 char temp = passChars[i];
135 passChars[i] = passChars[j];
136 passChars[j] = temp;
137 }
138
139 return new String(passChars);
140 }

173 private static int[] computeLPSArray(String pattern) {
174 int m = pattern.length();
175 int[] lps = new int[m];
176 int len = 0; // Length of previous longest prefix suffix
177 int i = 1;
178 lps[0] = 0; // lps[0] is always 0
179
180 while (i < m) {
181 if (pattern.charAt(i) == pattern.charAt(len)) {
182 len++;
183 lps[i] = len;
184 i++;
185 } else {
186 if (len != 0) {
187 len = lps[len - 1];
188 } else {
189 lps[i] = 0;
190 i++;
191 }
192 }
193 }
194
195 return lps;

210 public static int kmpSearch(String text, String pattern) {
211 if (text == null || pattern == null || pattern.isEmpty()) {
212 return -1;
213 }
214
215 int n = text.length();
216 int m = pattern.length();
217
218 if (m > n) {
219 return -1;
220 }
221
222 int[] lps = computeLPSArray(pattern);
223 int i = 0; // Index for text
224 int j = 0; // Index for pattern
225
226 while (i < n) {
227 if (pattern.charAt(j) == text.charAt(i)) {
228 i++;
229 j++;
230 }
231
232 if (j == m) {
233 return i - j; // Pattern found at index i-j
234 } else if (i < n && pattern.charAt(j) != text.charAt(i)) {
235 if (j != 0) {
236 j = lps[j - 1];
237 } else {
238 i++;
239 }
240 }
241 }
242
243 return -1; // Pattern not found
244 }

264 public static boolean isWeakPattern(String password) {
265 if (password == null || password.isEmpty()) {
266 return true;
267 }
268
269 // Common weak patterns
270 String[] weakPatterns = {
271 "123", "1234", "12345", "123456",
272 "abc", "abcd", "abcde",
273 "password", "pass", "pwd",
274 "qwerty", "admin", "user",
275 "000", "111", "aaa",
276 "password123", "admin123"
277 };
278 // Check case-insensitive
279 String lowerPassword = password.toLowerCase();
280
281 for (String pattern : weakPatterns) {
282 if (containsPattern(lowerPassword, pattern)) {
283 return true;
284 }
285 }
286
287 // Check for sequential numbers (ascending)
288 for (int i = 0; i < password.length() - 2; i++) {
289 if (Character.isDigit(password.charAt(i)) &&
290 Character.isDigit(password.charAt(i + 1)) &&
291 Character.isDigit(password.charAt(i + 2))) {
292 int d1 = password.charAt(i) - '0';
293 int d2 = password.charAt(i + 1) - '0';
294 int d3 = password.charAt(i + 2) - '0';
295
296 if (d2 == d1 + 1 && d3 == d2 + 1) {
297 return true; // Sequential like 123, 456, 789
298 }
299 }
300 }
301
302 // Check for repeated characters (3 or more)
303 for (int i = 0; i < password.length() - 2; i++) {
304 if (password.charAt(i) == password.charAt(i + 1) &&
305 password.charAt(i) == password.charAt(i + 2)) {
306 return true; // Repeated like aaa, 111
307 }
308 }
309
310 return false;
311 }

385 public static HuffmanNode buildHuffmanTree(Map<Character, Integer> frequencies) {
386 if (frequencies == null || frequencies.isEmpty()) {
387 return null;
388 }
389
390 PriorityQueue<HuffmanNode> pq = new PriorityQueue<>();
391
392 // Create leaf nodes for each character
393 for (Map.Entry<Character, Integer> entry : frequencies.entrySet()) {
394 pq.offer(new HuffmanNode(entry.getKey(), entry.getValue()));
395 }
396
397 // Build tree by combining nodes
398 while (pq.size() > 1) {
399 HuffmanNode left = pq.poll();
400 HuffmanNode right = pq.poll();
401 HuffmanNode parent = new HuffmanNode(
402 left.frequency + right.frequency, left, right
403 );
404 pq.offer(parent);
405 }
406
407 return pq.poll();
408 }

495 public static String decompressPassword(String compressed, HuffmanNode root) {
496 if (compressed == null || compressed.isEmpty() || root == null) {
497 return "";
498 }
499
500 StringBuilder decompressed = new StringBuilder();
501
502 // Special case: single character tree (root is a leaf)
503 if (root.isLeaf()) {
504 // Each bit represents one occurrence of the character
505 for (int i = 0; i < compressed.length(); i++) {
506 decompressed.append(root.character);
507 }
508
509 return decompressed.toString();
510 }
511
512 HuffmanNode current = root;
513
514 for (char bit : compressed.toCharArray()) {
515 if (bit == '0') {
516 current = current.left;
517 } else if (bit == '1') {
518 current = current.right;
519 }
520
521 if (current != null && current.isLeaf()) {
522 decompressed.append(current.character);
523 current = root;
524 }
